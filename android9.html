Chapter 9: Networking & APIs

Android Networking Fundamentals

Android provides several ways to handle network operations: HttpURLConnection, Volley, Retrofit, and OkHttp. Always perform network operations on background threads to avoid blocking the UI.

Basic HTTP Request with HttpURLConnection

public class NetworkUtils {
public static String makeHttpRequest(URL url) throws IOException {
HttpURLConnection connection = null;
InputStream inputStream = null;

try {
connection = (HttpURLConnection) url.openConnection();
connection.setRequestMethod("GET");
connection.setReadTimeout(10000);
connection.setConnectTimeout(15000);
connection.connect();

int responseCode = connection.getResponseCode();
if (responseCode != 200) {
throw new IOException("HTTP error code: " + responseCode);
}

inputStream = connection.getInputStream();
return readFromStream(inputStream);

} finally {
if (connection != null) {
connection.disconnect();
}
if (inputStream != null) {
inputStream.close();
}
}
}

private static String readFromStream(InputStream inputStream) throws IOException {
StringBuilder output = new StringBuilder();
if (inputStream != null) {
InputStreamReader inputStreamReader = new InputStreamReader(inputStream, Charset.forName("UTF-8"));
BufferedReader reader = new BufferedReader(inputStreamReader);
String line = reader.readLine();
while (line != null) {
output.append(line);
line = reader.readLine();
}
}
return output.toString();
}
}

// Usage in AsyncTask (deprecated, use other methods)
private class NetworkTask extends AsyncTask<String, Void, String> {
@Override
protected String doInBackground(String... urls) {
try {
URL url = new URL(urls[0]);
return NetworkUtils.makeHttpRequest(url);
} catch (IOException e) {
e.printStackTrace();
return null;
}
}

// Handle the result

return instance;

Retrofit Library (Most Popular)

// Add to build.gradle
dependencies {
implementation 'com.squareup.retrofit2:retrofit:2.9.0'
implementation 'com.squareup.retrofit2:converter-gson:2.9.0'
}

// Define API interface
public interface ApiService {
@GET("users/{user}")
Call<User> getUser(@Path("user") String user);

@GET("users")
Call<List<User>> getUsers();

@POST("users")
Call<User> createUser(@Body User user);

@PUT("users/{id}")
Call<User> updateUser(@Path("id") int id, @Body User user);

@DELETE("users/{id}")
Call<Void> deleteUser(@Path("id") int id);
}

// User model
public class User {
private int id;
private String name;
private String email;

// Getters and setters
}

// Create Retrofit instance
public class ApiClient {
private static final String BASE_URL = "https://jsonplaceholder.typicode.com/";
private static Retrofit retrofit = null;

public static Retrofit getClient() {
if (retrofit == null) {
retrofit = new Retrofit.Builder()
.baseUrl(BASE_URL)
.addConverterFactory(GsonConverterFactory.create())
.build();
}
return retrofit;
}
}

// Usage
ApiService apiService = ApiClient.getClient().create(ApiService.class);

// GET request
Call<User> call = apiService.getUser("1");
call.enqueue(new Callback<User>() {
@Override
public void onResponse(Call<User> call, Response<User> response) {
if (response.isSuccessful()) {
User user = response.body();
// Handle success
}
}

@Override
public void onFailure(Call<User> call, Throwable t) {
// Handle failure
Log.e("Error", t.getMessage());
}
});

Network Security Best Practices

Always use HTTPS in production, validate SSL certificates, and handle network errors gracefully.

// Custom OkHttpClient with SSL pinning (for security)
OkHttpClient.Builder builder = new OkHttpClient.Builder();

// Add certificate pinning
CertificatePinner certificatePinner = new CertificatePinner.Builder()
.add("api.example.com", "sha256/AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA=")
.build();

builder.certificatePinner(certificatePinner);

// Add interceptors for logging
HttpLoggingInterceptor logging = new HttpLoggingInterceptor();
logging.setLevel(HttpLoggingInterceptor.Level.BODY);
builder.addInterceptor(logging);

// Add timeout
builder.connectTimeout(30, TimeUnit.SECONDS);
builder.readTimeout(30, TimeUnit.SECONDS);

OkHttpClient client = builder.build();

// Use with Retrofit
Retrofit retrofit = new Retrofit.Builder()
.baseUrl(BASE_URL)
.client(client)
.addConverterFactory(GsonConverterFactory.create())
.build();

Handling Network State

public class NetworkUtils {
public static boolean isNetworkAvailable(Context context) {
ConnectivityManager connectivityManager =
(ConnectivityManager) context.getSystemService(Context.CONNECTIVITY_SERVICE);

if (connectivityManager != null) {
NetworkInfo activeNetworkInfo = connectivityManager.getActiveNetworkInfo();
return activeNetworkInfo != null && activeNetworkInfo.isConnected();
}
return false;
}

public static boolean isWifiConnected(Context context) {
ConnectivityManager connectivityManager =
(ConnectivityManager) context.getSystemService(Context.CONNECTIVITY_SERVICE);

NetworkInfo wifiNetwork = connectivityManager.getNetworkInfo(ConnectivityManager.TYPE_WIFI);
return wifiNetwork != null && wifiNetwork.isConnected();
}
}

// Usage
if (NetworkUtils.isNetworkAvailable(this)) {
// Perform network operation
} else {
// Show no network message
Toast.makeText(this, "No internet connection", Toast.LENGTH_SHORT).show();
}

AsyncTask vs Other Approaches

AsyncTask is deprecated. Use Volley, Retrofit, or Kotlin Coroutines for modern Android development.

Network Threading Best Practices

  • Never perform network operations on the main thread
  • Use background threads or AsyncTask alternatives
  • Handle configuration changes properly
  • Cancel network requests when appropriate
  • Show loading indicators to users
