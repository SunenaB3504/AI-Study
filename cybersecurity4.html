🛡️ Security Frameworks & Standards

OWASP Top 10

Most critical web application security risks. Updated regularly to reflect current threat landscape. Essential for web developers.

NIST Cybersecurity Framework

Comprehensive framework for improving cybersecurity. Covers identification, protection, detection, response, and recovery.

ISO 27001

International standard for information security management systems. Provides systematic approach to managing sensitive information.

Zero Trust Architecture

Security model that assumes no user or device is trusted by default. Requires continuous verification and least privilege access.

Compliance Frameworks

GDPR

EU General Data Protection Regulation. Focuses on personal data protection and privacy rights.

HIPAA

Health Insurance Portability and Accountability Act. Protects patient health information.

PCI DSS

Payment Card Industry Data Security Standard. Secures credit card transactions and data.

SOX

Sarbanes-Oxley Act. Ensures financial data accuracy and prevents fraud.

🧪 Security Testing & Vulnerability Assessment

Security Testing Phases

Planning

Define scope, objectives, and testing methodology. Identify systems and data to be tested.

Discovery

Map applications, identify entry points, and gather information about the target systems.

Assessment

Execute security tests, identify vulnerabilities, and assess their severity and impact.

Reporting

Document findings, provide remediation recommendations, and create executive summary.

Remediation

Fix identified vulnerabilities and retest to ensure proper resolution.

Monitoring

Continuous monitoring and periodic retesting to maintain security posture.

Automated Security Testing

// Security testing with OWASP ZAP API
const axios = require('axios');

class SecurityScanner {
    constructor(zapUrl = 'http://localhost:8080') {
        this.zapUrl = zapUrl;
        this.apiKey = process.env.ZAP_API_KEY;
    }

    async startScan(targetUrl) {
        try {
            // Start active scan
            const response = await axios.get(`${this.zapUrl}/JSON/ascan/action/scan/`, {
                params: {
                    url: targetUrl,
                    recurse: true,
                    inScopeOnly: true,
                    apikey: this.apiKey
                }
            });

            const scanId = response.data.scan;
            console.log(`Started security scan with ID: ${scanId}`);

            return scanId;
        } catch (error) {
            console.error('Failed to start security scan:', error.message);
            throw error;
        }
    }

    async getScanStatus(scanId) {
        try {
            const response = await axios.get(`${this.zapUrl}/JSON/ascan/view/status/`, {
                params: {
                    scanId: scanId,
                    apikey: this.apiKey
                }
            });

            return parseInt(response.data.status);
        } catch (error) {
            console.error('Failed to get scan status:', error.message);
            return 0;
        }
    }

    async getScanResults(scanId) {
        try {
            const response = await axios.get(`${this.zapUrl}/JSON/ascan/view/scanProgress/`, {
                params: {
                    scanId: scanId,
                    apikey: this.apiKey
                }
            });

            return response.data.scanProgress;
        } catch (error) {
            console.error('Failed to get scan results:', error.message);
            return [];
        }
    }

    async getAlerts() {
        try {
            const response = await axios.get(`${this.zapUrl}/JSON/core/view/alerts/`, {
                params: {
                    baseurl: '',
                    apikey: this.apiKey
                }
            });

            return response.data.alerts;
        } catch (error) {
            console.error('Failed to get alerts:', error.message);
            return [];
        }
    }

    async generateReport() {
        try {
            const response = await axios.get(`${this.zapUrl}/JSON/reports/action/generate/`, {
                params: {
                    title: 'Security Scan Report',
                    template: 'traditional-html',
                    apikey: this.apiKey
                }
            });

            return response.data;
        } catch (error) {
            console.error('Failed to generate report:', error.message);
            throw error;
        }
    }
}

        };

        return findings;
    }

    static async scanDependencies(packageJsonPath) {
        // Check for vulnerable dependencies using npm audit
        const { exec } = require('child_process');
        const { promisify } = require('util');
        const execAsync = promisify(exec);

        try {
            const { stdout } = await execAsync('npm audit --json', {
                cwd: require('path').dirname(packageJsonPath)
            });

            const auditResult = JSON.parse(stdout);
            return auditResult.vulnerabilities || {};
        } catch (error) {
            console.error('Failed to scan dependencies:', error.message);
            return {};
        }
    }
}

// Usage example
async function performSecurityAssessment(targetUrl, codebasePath) {
    console.log('Starting comprehensive security assessment...');

    // Dynamic Application Security Testing (DAST)
    const scanner = new SecurityScanner();
    const scanId = await scanner.startScan(targetUrl);

    // Wait for scan to complete
    let status = 0;
    do {
        status = await scanner.getScanStatus(scanId);
        console.log(`Scan progress: ${status}%`);
        await new Promise(resolve => setTimeout(resolve, 5000));
    } while (status < 100);

    // Get results
    const alerts = await scanner.getAlerts();
    console.log(`Found ${alerts.length} security alerts`);

    // Static Application Security Testing (SAST)
    const sastFindings = await SASTScanner.scanCodebase(codebasePath);
    console.log(`Found ${sastFindings.length} SAST findings`);

    // Dependency scanning
    const vulnerabilities = await SASTScanner.scanDependencies(codebasePath);
    console.log(`Found ${Object.keys(vulnerabilities).length} vulnerable dependencies`);

    // Generate comprehensive report
    const report = {
        timestamp: new Date().toISOString(),
        target: targetUrl,
        dast: {
            alerts: alerts.length,
            highRisk: alerts.filter(a => a.risk === 'High').length,
            mediumRisk: alerts.filter(a => a.risk === 'Medium').length
        },
        sast: {
            findings: sastFindings.length
        },
        dependencies: {
            vulnerabilities: Object.keys(vulnerabilities).length
        },
        recommendations: generateRecommendations(alerts, sastFindings, vulnerabilities)
    };

    return report;
}

function generateRecommendations(alerts, sastFindings, vulnerabilities) {
    const recommendations = [];

    if (alerts.some(a => a.name.includes('SQL Injection'))) {
        recommendations.push('Implement parameterized queries and input validation');
    }

    if (alerts.some(a => a.name.includes('XSS'))) {
        recommendations.push('Implement Content Security Policy and output encoding');
    }

    if (Object.keys(vulnerabilities).length > 0) {
        recommendations.push('Update vulnerable dependencies to latest secure versions');
    }

    return recommendations;
}

🚨 Incident Response & Recovery

Incident Response Process (NIST SP 800-61)

Preparation

Establish incident response team, develop policies, and prepare tools and resources.

Identification

Detect and assess potential security incidents. Determine scope and impact.

Containment

Stop the incident from spreading. Implement short-term and long-term containment strategies.

Eradication

Remove root cause and prevent recurrence. Clean up affected systems.

Recovery

Restore systems to normal operation. Monitor for any remaining issues.

Lessons Learned

Document incident, analyze response effectiveness, and update procedures.

Incident Response Plan Template

// Incident Response Plan Template
class IncidentResponsePlan {
    constructor() {
        this.incidentTypes = {
            MALWARE: 'Malware Infection',
            DDOS: 'Distributed Denial of Service',
            BREACH: 'Data Breach',
            INSIDER: 'Insider Threat',
            PHISHING: 'Phishing Attack'
        };

        this.severityLevels = {
            CRITICAL: { responseTime: 'Immediate (< 1 hour)', escalation: 'Executive + Legal' },
            HIGH: { responseTime: '< 4 hours', escalation: 'Security Team Lead' },
            MEDIUM: { responseTime: '< 24 hours', escalation: 'Department Head' },
            LOW: { responseTime: '< 72 hours', escalation: 'Team Lead' }
        };
    }

    assessSeverity(incident) {
        let severity = 'LOW';

        // Critical indicators
        if (incident.dataBreach || incident.customerImpact === 'HIGH') {
            severity = 'CRITICAL';
        }
        // High indicators
        else if (incident.systemDown || incident.financialImpact > 50000) {
            severity = 'HIGH';
        }
        // Medium indicators
        else if (incident.userImpact > 100 || incident.reputationRisk === 'MODERATE') {
            severity = 'MEDIUM';
        }

        return severity;
    }

    getResponseActions(incidentType, severity) {
        const actions = {
            MALWARE: {
                CRITICAL: [
                    'Isolate affected systems immediately',
                    'Engage forensic team',
                    'Notify executive leadership',
                    'Consider legal notification requirements'
                ],
                HIGH: [
                    'Quarantine affected systems',
                    'Run malware scans on network',
                    'Change all admin passwords',
                    'Monitor for lateral movement'
                ]
            },
            BREACH: {
                CRITICAL: [
                    'Stop data exfiltration',
                    'Assess breach scope and data types',
                    'Notify affected individuals',
                    'Engage legal counsel and PR team'
                ]
            }
        };

        return actions[incidentType]?.[severity] || ['Investigate incident', 'Document findings'];
    }

    generateIncidentReport(incident) {
        return {
            incidentId: `INC-${Date.now()}`,
            timestamp: new Date().toISOString(),
            type: incident.type,
            severity: this.assessSeverity(incident),
            description: incident.description,
            affectedSystems: incident.affectedSystems,
            impact: {
                users: incident.affectedUsers,
                financial: incident.financialImpact,
                operational: incident.systemsDown
            },
            response: {
                actions: this.getResponseActions(incident.type, this.assessSeverity(incident)),
                timeline: incident.responseTimeline,
                responsible: incident.assignedTeam
            },
            lessonsLearned: incident.lessonsLearned || []
        };
    }
}

// Automated Incident Detection
class IncidentDetector {
    constructor() {
        this.thresholds = {
            failedLogins: 10, // per minute
            unusualTraffic: 1000, // requests per minute
            errorRate: 0.05, // 5% error rate
            cpuUsage: 0.90 // 90% CPU usage
        };

        this.alerts = [];
    }

    monitorMetrics(metrics) {
        const alerts = [];

        // Check login failures
        if (metrics.failedLogins > this.thresholds.failedLogins) {
            alerts.push({
                type: 'BRUTE_FORCE',
                severity: 'HIGH',
                message: `High number of failed logins: ${metrics.failedLogins}/min`,
                timestamp: new Date()
            });
        }

        // Check traffic anomalies
        if (metrics.requestsPerMinute > this.thresholds.unusualTraffic) {
            alerts.push({
                type: 'DOS_ATTACK',
                severity: 'CRITICAL',
                message: `Unusual traffic spike: ${metrics.requestsPerMinute} req/min`,
                timestamp: new Date()
            });
        }

        // Check error rates
        if (metrics.errorRate > this.thresholds.errorRate) {
            alerts.push({
                type: 'SYSTEM_ERROR',
                severity: 'MEDIUM',
                message: `High error rate: ${(metrics.errorRate * 100).toFixed(1)}%`,
                timestamp: new Date()
            });
        }

        // Check system resources
        if (metrics.cpuUsage > this.thresholds.cpuUsage) {
            alerts.push({
                type: 'RESOURCE_EXHAUSTION',
                severity: 'HIGH',
                message: `High CPU usage: ${(metrics.cpuUsage * 100).toFixed(1)}%`,
                timestamp: new Date()
            });
        }

        this.alerts.push(...alerts);
        return alerts;
    }

    getActiveAlerts() {
        return this.alerts.filter(alert =>
            (new Date() - alert.timestamp) < (24 * 60 * 60 * 1000) // Last 24 hours
        );

}

🔄 DevSecOps: Integrating Security into Development

DevSecOps Best Practices

Shift Left Security

Integrate security early in the development lifecycle. Perform security testing during development, not just at the end.

Automated Security Testing

Implement automated security scans in CI/CD pipelines. Include SAST, DAST, dependency scanning, and container security.

Infrastructure as Code Security

Apply security principles to infrastructure code. Use tools like Terraform scanners and CloudFormation security checks.

Security Training

Provide ongoing security training for development teams. Include secure coding practices and threat awareness.

Continuous Monitoring

Implement continuous security monitoring and alerting. Use SIEM systems and security dashboards.

Compliance Automation

Automate compliance checks and reporting. Use policy-as-code tools to enforce security standards.

# GitHub Actions Security Pipeline
name: Security Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  security-scan:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout code
      uses: actions/checkout@v3

    - name: Run SAST (Static Application Security Testing)
      uses: github/super-linter/slim@v4
      env:
        DEFAULT_BRANCH: main
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

    - name: Dependency vulnerability scan
      run: |
        npm audit --audit-level=moderate
        pip-audit --format json | jq .

    - name: Container security scan
      uses: aquasecurity/trivy-action@master
      with:
        scan-type: 'fs'
        scan-ref: '.'
        format: 'sarif'
        output: 'trivy-results.sarif'

    - name: Upload Trivy scan results
      uses: github/codeql-action/upload-sarif@v2
      if: always()
      with:
        sarif_file: 'trivy-results.sarif'

    - name: Run DAST (Dynamic Application Security Testing)

        # Start application
