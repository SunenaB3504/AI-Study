4.1 Introduction to JavaScript

JavaScript is a high-level, interpreted programming language that enables interactive web pages. It runs in the browser and allows you to create dynamic content, control multimedia, animate images, and much more.

Key Concept: Client-Side Scripting

JavaScript executes in the user's browser, providing immediate feedback and reducing server requests.

JavaScript Evolution

  • 1995 Created by Brendan Eich at Netscape
  • 1997 ECMAScript 1 (ES1) standardization
  • 2009 ECMAScript 5 (ES5) with major improvements
  • 2015 ECMAScript 2015 (ES6) with modern features
  • 2023 ECMAScript 2023 (ES14) latest features

Adding JavaScript to HTML

There are several ways to include JavaScript in HTML:

1. Inline Scripts

<button onclick="alert('Hello!')">Click me</button>

2. Internal Scripts

<script>
    function greet() {
        alert('Hello from JavaScript!');
    }
</script>

3. External Scripts (Recommended)

<script src="script.js"></script>

4.2 Variables and Data Types

JavaScript variables store data values that can be referenced and manipulated throughout your code.

Variable Declaration

// Modern way (ES6+)
let name = "John";
const age = 25;

// Older way (still works)
var oldWay = "deprecated";

// Variable naming rules
let userName = "john_doe";  // camelCase
let $special = "allowed";   // $ and _ allowed
let _private = "convention"; // _ prefix for private

// Invalid names
// let 123invalid;  // Cannot start with number
// let user-name;   // Hyphens not allowed
// let let;         // Reserved keyword

Data Types

JavaScript has dynamic typing - variables can hold different types of data:

Data Type Example Description
String "Hello" Text data
Number 42 Integers and floats
Boolean true true or false
Undefined undefined Unassigned variable
Null null Intentionally empty
Object {} Complex data structures
Array [] Ordered collections

Type Checking and Conversion

// Type checking
typeof "Hello";        // "string"
typeof 42;            // "number"
typeof true;          // "boolean"
typeof undefined;     // "undefined"
typeof null;          // "object" (quirk!)
typeof {};            // "object"
typeof [];            // "object"

// Type conversion
String(42);           // "42"
Number("42");         // 42
Boolean(1);           // true
Boolean(0);           // false

// Automatic conversion
"5" + 3;              // "53" (string concatenation)
"5" - 3;              // 2 (numeric subtraction)

4.3 Operators

Operators perform operations on variables and values.

Arithmetic Operators

let x = 10;
let y = 3;

x + y;   // 13 (addition)
x - y;   // 7 (subtraction)
x * y;   // 30 (multiplication)
x / y;   // 3.333... (division)
x % y;   // 1 (modulus)
x ** y;  // 1000 (exponentiation)

// Increment/Decrement
x++;     // x = x + 1
x--;     // x = x - 1
++x;     // x = x + 1 (pre-increment)
--x;     // x = x - 1 (pre-decrement)

Comparison Operators

let a = 5;
let b = "5";

a == b;   // true (loose equality)
a === b;  // false (strict equality)
a != b;   // false (loose inequality)
a !== b;  // true (strict inequality)

a > b;    // false
a < b;    // false
a >= b;   // true
a <= b;   // true

Logical Operators

let x = true;
let y = false;

x && y;   // false (AND)
x || y;   // true (OR)
!x;       // false (NOT)

// Short-circuit evaluation
true && console.log("This runs");   // Logs message
false && console.log("This doesn't"); // Doesn't log

true || console.log("This doesn't");   // Doesn't log
false || console.log("This runs");     // Logs message

4.4 Control Flow

Control flow statements determine the execution path of your code.

Conditional Statements

// if-else
let age = 18;
if (age >= 18) {
    console.log("You can vote!");
} else {
    console.log("You cannot vote yet.");
}

// if-else if-else
let score = 85;
if (score >= 90) {
    console.log("A");
} else if (score >= 80) {
    console.log("B");
} else if (score >= 70) {
    console.log("C");
} else {
    console.log("F");
}

// Ternary operator
let canVote = age >= 18 ? "Yes" : "No";

// Switch statement
let day = "Monday";
switch (day) {
    case "Monday":
        console.log("Start of work week");
        break;
    case "Friday":
        console.log("TGIF!");
        break;
    case "Saturday":
    case "Sunday":
        console.log("Weekend!");
        break;
    default:
        console.log("Regular day");
}

Loops

// for loop
for (let i = 0; i < 5; i++) {
    console.log(i); // 0, 1, 2, 3, 4
}

// while loop
let count = 0;
while (count < 5) {
    console.log(count);
    count++;
}

// do-while loop
let num = 0;
do {
    console.log(num);
    num++;
} while (num < 5);

// for-of loop (arrays)
let fruits = ["apple", "banana", "orange"];
for (let fruit of fruits) {
    console.log(fruit);
}

// for-in loop (objects)
let person = {name: "John", age: 30};
for (let key in person) {
    console.log(key + ": " + person[key]);
}

4.5 Functions

Functions are reusable blocks of code that perform specific tasks.

Function Declaration

// Function declaration
function greet(name) {
    return "Hello, " + name + "!";
}

// Function expression
const sayHello = function(name) {
    return "Hello, " + name + "!";
};

// Arrow function (ES6+)
const greetArrow = (name) => {
    return "Hello, " + name + "!";
};

// Shorter arrow function
const greetShort = name => "Hello, " + name + "!";

// Calling functions
console.log(greet("John"));        // "Hello, John!"
console.log(sayHello("Jane"));     // "Hello, Jane!"
console.log(greetArrow("Bob"));    // "Hello, Bob!"
console.log(greetShort("Alice"));  // "Hello, Alice!"

Parameters and Arguments

// Default parameters
function greet(name = "Guest") {
    return "Hello, " + name + "!";
}

// Rest parameters
function sum(...numbers) {
    let total = 0;
    for (let num of numbers) {
        total += num;
    }
    return total;
}

console.log(sum(1, 2, 3, 4)); // 10

// Function as parameter
function operate(a, b, operation) {
    return operation(a, b);
}

const add = (a, b) => a + b;
const multiply = (a, b) => a * b;

console.log(operate(5, 3, add));      // 8
console.log(operate(5, 3, multiply)); // 15

4.6 Arrays

Arrays are ordered collections of values that can be of any type.

Creating Arrays

// Array literal
let fruits = ["apple", "banana", "orange"];

// Array constructor
let numbers = new Array(1, 2, 3, 4, 5);

// Mixed types
let mixed = [1, "hello", true, null, {name: "John"}];

// Empty array
let empty = [];

Array Methods

let fruits = ["apple", "banana", "orange"];

// Adding elements
fruits.push("grape");        // Add to end
fruits.unshift("mango");     // Add to beginning

// Removing elements
let last = fruits.pop();     // Remove from end
let first = fruits.shift();  // Remove from beginning

// Finding elements
let index = fruits.indexOf("banana");  // 1
let hasOrange = fruits.includes("orange"); // true

// Slicing and splicing
let citrus = fruits.slice(1, 3);  // ["banana", "orange"]
fruits.splice(1, 1, "kiwi");     // Replace 1 element at index 1

// Iterating
fruits.forEach(fruit => console.log(fruit));

let upperFruits = fruits.map(fruit => fruit.toUpperCase());
let longFruits = fruits.filter(fruit => fruit.length > 5);

4.7 Objects

Objects are collections of key-value pairs that represent complex data structures.

Creating Objects

// Object literal
let person = {
    name: "John Doe",
    age: 30,
    email: "john@example.com",
    hobbies: ["reading", "coding"],
    address: {
        street: "123 Main St",
        city: "Anytown",
        zipCode: "12345"
    }
};

// Constructor function
function Person(name, age) {
    this.name = name;
    this.age = age;
    this.greet = function() {
        return "Hello, I'm " + this.name;
    };
}

let john = new Person("John", 30);

// ES6 Classes
class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }

    greet() {
        return "Hello, I'm " + this.name;
    }
}

Working with Objects

// Accessing properties
console.log(person.name);        // "John Doe"
console.log(person["age"]);      // 30

// Adding properties
person.job = "Developer";

// Modifying properties
person.age = 31;

// Deleting properties
delete person.email;

// Checking properties
console.log("name" in person);    // true
console.log(person.hasOwnProperty("name")); // true

// Object methods
let keys = Object.keys(person);
let values = Object.values(person);
let entries = Object.entries(person);

4.8 DOM Manipulation

The Document Object Model (DOM) represents the HTML document as a tree structure that JavaScript can manipulate.

Selecting Elements

// By ID
let header = document.getElementById("main-header");

// By class name
let buttons = document.getElementsByClassName("btn");

// By tag name
let paragraphs = document.getElementsByTagName("p");

// By CSS selector (modern way)
let firstButton = document.querySelector(".btn");
let allButtons = document.querySelectorAll(".btn");

Modifying Elements

let element = document.getElementById("myElement");

// Change text content
element.textContent = "New text";

// Change HTML content
element.innerHTML = "<strong>Bold text</strong>";

// Change attributes
element.setAttribute("class", "highlight");
element.className = "highlight"; // Alternative

// Change styles
element.style.color = "red";
element.style.fontSize = "20px";

// Add/remove classes
element.classList.add("active");
element.classList.remove("hidden");
element.classList.toggle("visible");

Creating and Removing Elements

// Create element
let newDiv = document.createElement("div");
newDiv.textContent = "New element";
newDiv.className = "new-item";

// Add to document
document.body.appendChild(newDiv);

// Insert before another element
let reference = document.getElementById("reference");
document.body.insertBefore(newDiv, reference);

// Remove element
let toRemove = document.getElementById("remove-me");
toRemove.parentNode.removeChild(toRemove);

// Modern way (if supported)
toRemove.remove();

4.9 Event Handling

Events allow JavaScript to respond to user interactions and other occurrences.

Event Listeners

let button = document.getElementById("myButton");

// Traditional way
button.onclick = function() {
    console.log("Button clicked!");
};

// Modern way (recommended)
button.addEventListener("click", function(event) {
    console.log("Button clicked!");
    console.log("Event type:", event.type);
    console.log("Target:", event.target);
});

// Arrow function
button.addEventListener("click", (event) => {
    console.log("Button clicked with arrow function!");
});

// Remove event listener
function handleClick() {
    console.log("Click handled");
}

button.addEventListener("click", handleClick);
// Later...
button.removeEventListener("click", handleClick);

Common Events

Event Description Example Use
click Mouse click Button interactions
mouseover Mouse enters element Hover effects
mouseout Mouse leaves element Hover effects
keydown Key pressed down Keyboard input
keyup Key released Keyboard input
submit Form submitted Form validation
load Page finished loading Initialize scripts
scroll Page or element scrolled Infinite scroll

4.10 Error Handling

JavaScript provides mechanisms to handle and recover from errors.

Try-Catch Blocks

try {
    // Code that might throw an error
    let result = riskyOperation();
    console.log("Success:", result);
} catch (error) {
    // Handle the error
    console.error("Error occurred:", error.message);
} finally {
    // Always executed
    console.log("Cleanup code here");
}

Throwing Errors

function divide(a, b) {
    if (b === 0) {
        throw new Error("Division by zero!");
    }
    return a / b;
}

try {
    let result = divide(10, 0);
} catch (error) {
    console.log(error.message); // "Division by zero!"
}
